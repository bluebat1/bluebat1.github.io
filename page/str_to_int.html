//字符串轉整數值 
//支持負數
int str_to_int(char *str, int len)
{
    int x=0;
    int i;
    if(str[0] == '-'){
        for(i=1; i <= len; i++){
            x *= 10;
            x += str[i]-'0';
        }
        x *= -1;
    }else{
        for(i=0; i < len; i++){
            x *= 10;
            x += str[i]-'0';
        }
    }
    return x;
}
